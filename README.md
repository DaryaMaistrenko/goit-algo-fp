def add_edges(graph, node, pos, x=0, y=0, layer=1):
  if node is not None:
    graph.add_node(node.id, color=node.color, label=node.val) # –í–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è id —Ç–∞ –∑–±–µ—Ä–µ–∂–µ–Ω–Ω—è –∑–Ω–∞—á–µ–Ω–Ω—è –≤—É–∑–ª–∞

      pos[node.right.id] = (r, y - 1)

  return graph

def draw_tree(tree_root):
